CREATE TABLE wiki_categorias (
    id          INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    proyecto_id INT UNSIGNED NOT NULL,
    nombre      VARCHAR(100) NOT NULL,
    color       VARCHAR(7) NOT NULL DEFAULT '#6366f1',
    creado_en   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_wc_proyecto FOREIGN KEY (proyecto_id) REFERENCES proyectos(id) ON DELETE CASCADE
) ENGINE=InnoDB;

CREATE TABLE wiki_paginas (
    id           INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    proyecto_id  INT UNSIGNED NOT NULL,
    categoria_id INT UNSIGNED NULL,
    titulo       VARCHAR(200) NOT NULL,
    contenido    LONGTEXT,
    creado_en    DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    actualizado_en DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    CONSTRAINT fk_wp_proyecto  FOREIGN KEY (proyecto_id)  REFERENCES proyectos(id)      ON DELETE CASCADE,
    CONSTRAINT fk_wp_categoria FOREIGN KEY (categoria_id) REFERENCES wiki_categorias(id) ON DELETE SET NULL
) ENGINE=InnoDB;
